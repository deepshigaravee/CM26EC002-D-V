Example using maven project
import reactor.core.publisher.Flux;
import java.util.Arrays;
import java.util.List;
 
public class ReactiveExample {
 
   public static void main(String[] args) {
      List<String> words = Arrays.asList("apple", "banana", "cherry", "date", "elderberry");
 
       // 1. Create a Flux (Publisher) from a list of items
      Flux<String> reactiveStream = Flux.fromIterable(words)
           // 2. Apply a filter operation (only words longer than 5 characters)
          .filter(word -> word.length() > 5)
           // 3. Apply a map operation (convert to uppercase)
          .map(String::toUpperCase)
           // 4. Log the data flow for demonstration/debugging
           .log();
 
       // 5. Subscribe to the stream to start the process and define what to do with each item
      reactiveStream.subscribe(
           item -> System.out.println("Subscriber received: " + item), // onNext consumer
           error -> System.err.println("Subscriber error: " + error),   // onError consumer
           () -> System.out.println("Subscriber completed")            // onComplete runnable
       );
   }
}
